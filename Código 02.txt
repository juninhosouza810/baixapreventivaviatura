    <script>
        function enviarEmail() {
            var nome = document.getElementById('nome').value;
            var re = document.getElementById('re').value;
            var unidade = document.getElementById('unidade').value;
            var mes = document.getElementById('mes').value;
            var confirmacao = document.getElementById('confirmacao');
            var successMessage = document.getElementById('success-message');
            
            if (!nome || !re) {
                alert("Por favor, preencha todos os campos!");
                return;
            }
            
            if (!confirmacao.checked) {
                alert("Você deve confirmar a declaração antes de enviar!");
                return;
            }
            
            var assunto = encodeURIComponent("PVT - " + nome);
            var corpo = encodeURIComponent("Nome: " + nome + "\nRE: " + re + "\nUnidade: " + unidade + "\nPVT Mês: " + mes + "\n\nDeclaro que efetuei o PVT referente ao mês indicado acima.");
            
            window.location.href = "mailto:2bpmi5cia@policiamilitar.sp.gov.br?subject=" + assunto + "&body=" + corpo;
            
            document.getElementById('nome').value = "";
            document.getElementById('re').value = "";
            document.getElementById('mes').selectedIndex = 0;
            confirmacao.checked = false;
            
            successMessage.style.display = "block";
        }
    </script>
